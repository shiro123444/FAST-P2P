Testing Toast Notification

Open the console and navigate to http://localhost:5174

Then run this in the console:

// Wait for page to load, create a room, then test copy
setTimeout(() => {
    // Click create button
    const createBtn = document.querySelector('.action-button-primary');
    if (createBtn) {
        createBtn.click();
        console.log('Clicked create button');
        
        // Wait for room to be created, then click copy
        setTimeout(() => {
            const copyBtn = document.querySelector('.text-button');
            if (copyBtn && !copyBtn.disabled) {
                copyBtn.click();
                console.log('Clicked copy button - Toast should appear!');
            }
        }, 2000);
    }
}, 1000);
